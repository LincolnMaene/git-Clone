import zlib

def readTree(fileHash):
    
    fileLoc = f".git/objects/{fileHash[:2]}/{fileHash[2:]}"
    
    with open(fileLoc, "rb") as f:
        
        decompressedData = zlib.decompress(f.read())
        _, treeData = decompressedData.split(b"\0", maxsplit = 1)
        
        while treeData:
            modeName, shaData = decompressedData.split(b"\0", maxsplit = 1)
            _, name = modeName.split()
            print(name.decode("utf-8"))
            treeData = shaData[20:]
